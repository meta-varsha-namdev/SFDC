public class FilterController 
{
    public string billingPickValue{get; set;}
    
    public List<Account> accountList{get; set;}
    public string inputString{get; set;}
    public List<SelectOption> getbillingOptions()
    {
        List<SelectOption> options = new List<selectoption>();
        options.add(new SelectOption('BillingCity','BillingCity'));
        options.add(new SelectOption('BillingState','BillingState'));
       	options.add(new SelectOption('BillingCountry','BillingCountry'));
        return options;
    }
    public List<Account> getaccountList()
    {
        System.debug('billingPickValue'+billingPickValue);
        System.debug('inputString'+inputString);
        accountList = Database.query('SELECT Name, ID, BillingCountry, BillingState, BillingCity FROM Account WHERE '+billingPickValue+' like \''+inputString+'\' Limit 100');
        return accountList;
    }
 
   
    public void showRecords()
    {
        getaccountList();
    }
    
}
